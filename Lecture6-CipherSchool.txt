2. GRADING :-

#include<iostream>
using namespace std;

int main()
{
    int marks;
    cout<<"Student, please enter the marks to calculate the grade : "<<endl;
    cin>>marks;

    if(marks>90)
    {
        cout<<"A";
    }
    else if(marks<90 && marks>80)
    {
        cout<<"B";
    }
    else if(marks<80 && marks>70)
    {
        cout<<"C";
    }
    else
    {
        cout<<"Pass";
    }
    return 0;
}

4. Switch Case :-

#include<iostream>
using namespace std;

int main()
{
    cout<<"Enter a integer a ";
    cin>>a;

    switch(a) 
    {
        case 1:
            cout<<"The value of a is 1 "<<endl;
            break;              

        case 2:
            cout<<"The value of a is 2 "<<endl;
            break;

        default:
            cout<<"default will be always printed"<<endl;
            break;
    }
  return 0;
}
